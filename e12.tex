\begin{frame}

\section*{LED}

\begin{frame}
    \frametitle{LED}
      	\begin{itemize}
			\item LED - light-emitting diode
			\item Leuchtet wenn in Durchlassrichtung betrieben
			\item Betriebsspannung je nach LED-Farbe von etwa $1.5V$ - $3.2V$
