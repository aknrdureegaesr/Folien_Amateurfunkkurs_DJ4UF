\section{Verstärker}
\begin{frame}
\frametitle{Verstärker}
\begin{center}
\begin{block}{Definition eines Verstärkers nach Captain Obvious}
  \begin{Large}
    Es ist nur dann eine Verstärkung, wenn die Leistung am Ausgang größer ist, als die am Eingang
  \end{Large}
\end{block}
\end{center}
\end{frame}
