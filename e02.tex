\begin{frame}
    \frametitle{Effektivwert vs Spitzen Wert}
    \begin{align} 
		\int \! sin^2(w t) \, \mathrm{d}t & = \frac{t}{2} - \frac{1}{4w}sin^2(2wt) + c \\
		U_{eff}^2 \cdot \int_0^T \! sin^2(w t) \, \mathrm{d}t & = \frac{T}{2} \cdot \frac{u_{spitze}^2}{T} \\
		U_{eff} & = \frac{1}{\sqrt{2}} u_{spitze}
	\end{align}
